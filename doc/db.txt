drop database IF EXISTS gradingsystem;

create database gradingsystem;

use gradingsystem;



CREATE TABLE `login` (

                         `loginid` int(11) NOT NULL AUTO_INCREMENT,

                         `username` varchar(45) DEFAULT NULL,

                         `password` varchar(45) DEFAULT NULL,

                         `numcourses` int(11),

                         `numcategories` int(11),

                         `numassignment` int(11),

                         `numstudent` int(11),

                         `numbonuspoints` int(11),

                         `numstudentcomments` int(11),

                         `numgrades` int(11),

                         PRIMARY KEY (`loginid`)

) ENGINE=InnoDB;

CREATE TABLE `course` (

                          `courseid` int(11) NOT NULL AUTO_INCREMENT,

                          `semester` varchar(45) DEFAULT NULL,

                          `name` varchar(45) DEFAULT NULL,

                          `sectionid` int(11),

                          PRIMARY KEY (`courseid`)

);

CREATE TABLE `category`(
                           `categoryid` int(11) NOT NULL AUTO_INCREMENT,
                           `courseid` int(11) NOT NULL,
                           `weight` double,
                           `name` varchar(45),
                           PRIMARY KEY (`categoryid`)
);

CREATE TABLE `assignment`(
                             `assignmentid` int(11) NOT NULL AUTO_INCREMENT,
                             `categoryid` int(11) NOT NULL,
                             `weight` double,
                             `releasedate` DATE,
                             `duedate`  DATE,
                             `maxpoints` double,
                             `name` varchar(45),
                             PRIMARY KEY (`assignmentid`)
);

CREATE TABLE `student`(
                          `studentid` int(11) NOT NULL AUTO_INCREMENT,
                          `courseid` int(11) NOT NULL,
                          `fname` varchar(45),
                          `mname` varchar(45),
                          `lname` varchar(45),
                          `email` varchar(45),
                          PRIMARY KEY (`studentid`)

);

CREATE TABLE `bonuspoints`(
                              `pointid` int(11),
                              `studentid` int(11),
                              `points` double,
                              primary key(`pointid`)
);

CREATE TABLE `studentcomments`(
                                  `commentid` int(11),
                                  `studentid` int(11),
                                  `comment` varchar(45),
                                  primary key(`commentid`)
);

CREATE TABLE `grade`(
                        `gradeid` int(11) NOT NULL AUTO_INCREMENT,
                        `rawscore` double,
                        `assignmentid` int(11),
                        `studentid` int(11),
                        `bonuspoints` double,
                        `lettergrade` varchar(45),
                        `comment` varchar(45),
                        primary key(`gradeid`)
);
